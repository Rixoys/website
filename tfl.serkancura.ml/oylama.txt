CREATE TABLE oylama (
    id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    baslik VARCHAR(255) NOT NULL
);

CREATE TABLE secenekler (
    id INT(11) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    oylama_id INT(11) UNSIGNED NOT NULL,
    adi VARCHAR(255) NOT NULL,
    oy_sayisi INT(11) UNSIGNED DEFAULT 0,
    FOREIGN KEY (oylama_id) REFERENCES oylama(id)
);
